fn validate(x: i32, limit: i32) -> bool {
    if x == limit { return true; }
    if x != limit { return false; }
    if x >= limit || x <= 0 { return false; }
    if x > 0 && x < limit { return true; }
    return x == 0;
}

    .filter(|x| x != 0)
